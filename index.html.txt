<script>
    // Configured with crisp, aesthetic stock photos for your specific catalog items
    const products = [
        { 
            id: 'tea', 
            name: 'Sun-Dried Lemongrass Tea', 
            desc: 'Whole, loose-leaf herbal cuts. Perfect for crisp morning detox brews and immunity boosts.',
            price: 299, 
            unit: '100g Pack',
            img: 'https://images.unsplash.com/photo-1597318181409-cf64d0b5e8a2?auto=format&fit=crop&w=400&q=80'
        },
        { 
            id: 'oil', 
            name: 'Pure Lemongrass Essential Oil', 
            desc: '100% pure steam-distilled aromatic oil. Excellent therapeutic grade for diffusers and skin remedies.',
            price: 450, 
            unit: '15ml Bottle',
            img: 'https://images.unsplash.com/photo-1608571423902-eed4a5ad8108?auto=format&fit=crop&w=400&q=80'
        }
    ];

    let cart = {};

    function renderProducts() {
        const grid = document.getElementById('products');
        grid.innerHTML = products.map(p => `
            <div class="product-card">
                <div class="product-image" style="background-image: url('${p.img}')"></div>
                <div class="product-info">
                    <h3>${p.name}</h3>
                    <p class="product-description">${p.desc}</p>
                    <div class="price-row">
                        <div class="price">₹${p.price} <span style="font-size:0.85rem; font-weight:normal; color:#666;">/${p.unit}</span></div>
                        <div class="qty-controls">
                            <button class="qty-btn" onclick="updateQty('${p.id}', -1)">-</button>
                            <span class="qty-val" id="qty-${p.id}">0</span>
                            <button class="qty-btn" onclick="updateQty('${p.id}', 1)">+</button>
                        </div>
                    </div>
                </div>
            </div>
        `).join('');
    }

    function updateQty(id, change) {
        if (!cart[id]) cart[id] = 0;
        cart[id] += change;
        if (cart[id] < 0) cart[id] = 0;
        document.getElementById(`qty-${id}`).innerText = cart[id];
        calculateTotal();
    }

    function calculateTotal() {
        let subtotal = 0;
        products.forEach(p => {
            const qty = cart[p.id] || 0;
            subtotal += qty * p.price;
        });

        // Simple shipping calculation logic: ₹50 flat for ground shipping across India
        const shippingFee = subtotal > 0 ? 50 : 0;
        const grandTotal = subtotal + shippingFee;

        document.getElementById('subtotal').innerText = `₹${subtotal}`;
        document.getElementById('shipping-fee').innerText = `₹${shippingFee}`;
        document.getElementById('grand-total').innerText = `₹${grandTotal}`;

        const gatewayBox = document.getElementById('payment-button-container');
        
        if (subtotal > 0) {
            // INTEGRATION POINT: Replace "pl_YOUR_BUTTON_ID" with your real Approved Razorpay / Instamojo button script
            gatewayBox.innerHTML = `
                <form>
                    <script src="https://checkout.razorpay.com/v1/payment-button.js" 
                        data-payment_button_id="pl_YOUR_BUTTON_ID" async> 
                    </script> 
                </form>
            `;
        } else {
            gatewayBox.innerHTML = `<p style="color: #888; font-style: italic; font-size: 0.95rem;">Please add items to your cart to open secure payment gateway.</p>`;
        }
    }

    renderProducts();
</script>
